using Framework.Core;
using Script.Manager;
using TMPro;
using UnityEngine;

public class HintChangePanel:PanelBase
{
    private Transform hintText;
    private Transform hintKeyboard;
    private SingleSubtitleConfig curSubtitleConfig;
    public override void Init()
    {
        isCloseSelf = true;
        curSubtitleConfig = SubtitleManager.Instance.curSubtitleConfig;
        closeTime = curSubtitleConfig.NeedTimeToClose;
        base.Init();
    }

    public override void Bind()
    {
        hintText= TransformUtil.Find(PanelObj.transform, "HintText");
        hintKeyboard = TransformUtil.Find(PanelObj.transform, "HintKeyboard");
    }

    public override void Show()
    {
        hintText.GetComponent<TextMeshProUGUI>().text = curSubtitleConfig.Text;
        hintKeyboard.GetComponent<TextMeshProUGUI>().text = curSubtitleConfig.Text;
    }
}
